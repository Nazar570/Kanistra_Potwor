using System.Collections;
using System.Collections.Generic;
using TMPro;
using UnityEngine;
using UnityEngine.UI;

public class HintSystem : MonoBehaviour
{
    public static HintSystem Instance { get; private set; }

    [Header("UI")]
    [Tooltip("CanvasGroup ca³ego panelu podpowiedzi. Jeœli podpiête, alpha bêdzie sterowana przez CanvasGroup.")]
    [SerializeField] private CanvasGroup hintCanvasGroup;

    [Tooltip("Tekst TextMeshProUGUI. Jeœli u¿ywasz legacy Text, zostaw puste i podpnij legacyHintText.")]
    [SerializeField] private TextMeshProUGUI hintText;

    [Tooltip("Opcjonalny legacy Text, jeœli nie u¿ywasz TextMeshPro.")]
    [SerializeField] private Text legacyHintText;

    [Header("Domyœlne czasy")]
    [SerializeField] private float defaultFadeInDuration = 0.45f;
    [SerializeField] private float defaultVisibleDuration = 5f;
    [SerializeField] private float defaultFadeOutDuration = 0.35f;
    [SerializeField] private AnimationCurve fadeCurve = AnimationCurve.EaseInOut(0f, 0f, 1f, 1f);

    [Header("Start")]
    [SerializeField] private bool startHidden = true;

    private readonly Queue<HintMessage> hintQueue = new Queue<HintMessage>();
    private Coroutine activeRoutine;
    private bool hasPinnedObjective;
    private string pinnedObjectiveText = string.Empty;
    private string pendingObjectiveText = string.Empty;

    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Debug.LogWarning("[HintSystem] Wykryto wiêcej ni¿ jedn¹ instancjê. Ta zostanie zignorowana.", this);
            return;
        }

        Instance = this;

        if (startHidden)
            SetHintAlpha(0f);

        if (hintCanvasGroup != null)
        {
            hintCanvasGroup.interactable = false;
            hintCanvasGroup.blocksRaycasts = false;
        }
    }

    private void OnDestroy()
    {
        if (Instance == this)
            Instance = null;
    }

    public void ShowHint(string message, float visibleDuration = -1f, float fadeInDuration = -1f, float fadeOutDuration = -1f)
    {
        if (string.IsNullOrWhiteSpace(message))
            return;

        hintQueue.Enqueue(new HintMessage(
            message,
            visibleDuration < 0f ? defaultVisibleDuration : visibleDuration,
            fadeInDuration < 0f ? defaultFadeInDuration : fadeInDuration,
            fadeOutDuration < 0f ? defaultFadeOutDuration : fadeOutDuration
        ));

        if (activeRoutine == null)
            activeRoutine = StartCoroutine(ProcessHintQueueRoutine());
    }

    public void ShowObjective(string message, float fadeInDuration = -1f)
    {
        if (string.IsNullOrWhiteSpace(message))
            return;

        pinnedObjectiveText = message;
        pendingObjectiveText = message;
        hasPinnedObjective = true;

        if (activeRoutine == null)
            activeRoutine = StartCoroutine(FadeObjectiveRoutine(fadeInDuration < 0f ? defaultFadeInDuration : fadeInDuration));
    }

    public void ClearObjective()
    {
        hasPinnedObjective = false;
        pinnedObjectiveText = string.Empty;
        pendingObjectiveText = string.Empty;

        if (activeRoutine != null)
        {
            StopCoroutine(activeRoutine);
            activeRoutine = StartCoroutine(FadeOutAndHideRoutine(defaultFadeOutDuration));
            return;
        }

        if (hintQueue.Count == 0)
            activeRoutine = StartCoroutine(FadeOutAndHideRoutine(defaultFadeOutDuration));
    }

    public void HideAll()
    {
        if (activeRoutine != null)
        {
            StopCoroutine(activeRoutine);
            activeRoutine = null;
        }

        hintQueue.Clear();
        hasPinnedObjective = false;
        pinnedObjectiveText = string.Empty;
        pendingObjectiveText = string.Empty;

        SetHintText(string.Empty);
        SetHintAlpha(0f);
    }

    private IEnumerator ProcessHintQueueRoutine()
    {
        while (hintQueue.Count > 0)
        {
            HintMessage message = hintQueue.Dequeue();
            yield return DisplayTemporaryHintRoutine(message);
        }

        activeRoutine = null;
    }

    private IEnumerator FadeObjectiveRoutine(float fadeInDuration, bool clearActiveRoutineWhenDone = true)
    {
        if (hintCanvasGroup != null)
            hintCanvasGroup.gameObject.SetActive(true);

        if (!hasPinnedObjective || string.IsNullOrWhiteSpace(pinnedObjectiveText))
        {
            SetHintText(string.Empty);
            yield return FadeHintAlphaRoutine(GetCurrentAlpha(), 0f, defaultFadeOutDuration);

            if (clearActiveRoutineWhenDone)
                activeRoutine = null;

            yield break;
        }

        SetHintText(pinnedObjectiveText);
        yield return FadeHintAlphaRoutine(GetCurrentAlpha(), 1f, fadeInDuration);

        if (clearActiveRoutineWhenDone)
            activeRoutine = null;
    }

    private IEnumerator DisplayTemporaryHintRoutine(HintMessage message)
    {
        if (hintCanvasGroup != null)
            hintCanvasGroup.gameObject.SetActive(true);

        SetHintText(message.text);
        yield return FadeHintAlphaRoutine(GetCurrentAlpha(), 1f, message.fadeInDuration);

        if (message.visibleDuration > 0f)
            yield return new WaitForSeconds(message.visibleDuration);

        yield return FadeHintAlphaRoutine(GetCurrentAlpha(), 0f, message.fadeOutDuration);

        if (hasPinnedObjective)
        {
            pinnedObjectiveText = pendingObjectiveText;
            SetHintText(pinnedObjectiveText);
            yield return FadeHintAlphaRoutine(GetCurrentAlpha(), 1f, defaultFadeInDuration);
        }
    }

    private IEnumerator FadeOutAndHideRoutine(float fadeOutDuration)
    {
        yield return FadeHintAlphaRoutine(GetCurrentAlpha(), 0f, fadeOutDuration);
        SetHintText(string.Empty);

        activeRoutine = null;

        if (hintQueue.Count > 0)
            activeRoutine = StartCoroutine(ProcessHintQueueRoutine());
    }

    private IEnumerator FadeHintAlphaRoutine(float fromAlpha, float toAlpha, float duration)
    {
        fromAlpha = Mathf.Clamp01(fromAlpha);
        toAlpha = Mathf.Clamp01(toAlpha);
        duration = Mathf.Max(0f, duration);

        if (duration <= 0f)
        {
            SetHintAlpha(toAlpha);
            yield break;
        }

        float timer = 0f;

        while (timer < duration)
        {
            timer += Time.unscaledDeltaTime;
            float normalizedTime = Mathf.Clamp01(timer / duration);
            float easedTime = fadeCurve != null ? fadeCurve.Evaluate(normalizedTime) : normalizedTime;
            SetHintAlpha(Mathf.LerpUnclamped(fromAlpha, toAlpha, easedTime));
            yield return null;
        }

        SetHintAlpha(toAlpha);
    }

    private float GetCurrentAlpha()
    {
        if (hintCanvasGroup != null)
            return hintCanvasGroup.alpha;

        if (hintText != null)
            return hintText.color.a;

        if (legacyHintText != null)
            return legacyHintText.color.a;

        return 0f;
    }

    private string GetCurrentText()
    {
        if (hintText != null)
            return hintText.text;

        if (legacyHintText != null)
            return legacyHintText.text;

        return string.Empty;
    }

    private void SetHintText(string text)
    {
        if (hintText != null)
            hintText.text = text;

        if (legacyHintText != null)
            legacyHintText.text = text;
    }

    private void SetHintAlpha(float alpha)
    {
        alpha = Mathf.Clamp01(alpha);

        if (hintCanvasGroup != null)
        {
            hintCanvasGroup.alpha = alpha;
            return;
        }

        Color color;

        if (hintText != null)
        {
            color = hintText.color;
            color.a = alpha;
            hintText.color = color;
        }

        if (legacyHintText != null)
        {
            color = legacyHintText.color;
            color.a = alpha;
            legacyHintText.color = color;
        }
    }

    private struct HintMessage
    {
        public readonly string text;
        public readonly float visibleDuration;
        public readonly float fadeInDuration;
        public readonly float fadeOutDuration;

        public HintMessage(string text, float visibleDuration, float fadeInDuration, float fadeOutDuration)
        {
            this.text = text;
            this.visibleDuration = visibleDuration;
            this.fadeInDuration = fadeInDuration;
            this.fadeOutDuration = fadeOutDuration;
        }
    }
}

        Collider collider = GetComponent<Collider>();

    }

    private void OnTriggerEnter(Collider other)

        if (triggered || !IsPlayer(other))

        triggered = true;
        StartCoroutine(PlayEndingRoutine());
    }

    {
        if (playerController == null)
            playerController = FindObjectOfType<FirstPersonController>(true);

        if (characterController == null && playerController != null)
            characterController = playerController.GetComponent<CharacterController>();

        if (characterController == null)

        if (playerCameraRoot == null && playerController != null)
        {
            Transform foundCameraRoot = playerController.transform.Find("PlayerCameraRoot");

            if (foundCameraRoot != null)
                playerCameraRoot = foundCameraRoot;
        }

        if (fadeCanvas == null)

        if (endGameScreen == null)
            endGameScreen = FindObjectOfType<EndGameScreen>(true);
    }
